import { Injectable } from '@nestjs/common';
import { PrismaClient } from '@prisma/client';
import { ID_STARTERS, RETURN_DATA } from 'src/misc/parameterConstants';
import { ReturnMessage } from 'src/types/Database';
import { v4 as uuidv4 } from 'uuid';
const prisma = new PrismaClient();
@Injectable()
export class TimezoneService {
  async importTimezones(): Promise<ReturnMessage> {
    const timezones = this.getTimezones();

    try {
      for (let i = 0; i < timezones.length; i++) {
        await prisma.timezones.create({
          data: {
            timezoneUUID: `${ID_STARTERS.TIMEZONES}${uuidv4()}`,
            timezoneName: timezones[i],
          },
        });
      }
    } catch (error) {
      return RETURN_DATA.DATABASE_ERROR;
    }
    return RETURN_DATA.SUCCESS;
  }

  async getTimezoneByName(name: string): Promise<ReturnMessage> {
    try {
      await prisma.timezones.findFirst({
        where: {
          timezoneName: name,
        },
      });
      return RETURN_DATA.SUCCESS;
    } catch (error) {
      return RETURN_DATA.DATABASE_ERROR;
    }
  }
